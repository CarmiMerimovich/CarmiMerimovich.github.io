struct node {
    struct node *left;
    struct node *right;
    long long value;
};

void trav(struct node *node, (*func)()) {
    if (node == NULL)
        return (0);

    if (node->left == NULL  &&  node->right == NULL) {
        (*func)(node);
        return;
    }   

    trav(node->left);
    trav(node->right);
    return;
